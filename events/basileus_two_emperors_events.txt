#   .0007  War declared â€” the Basileus's court (CB on_declaration).

#   .0011  War lost / white peace â€” the dispute fails.
#   .0020  The demand for precedence â€” the Western emperor decides (AI).

#   .0030  Vindication â€” the rival empire fell by other hands.

# 0001 â€” A Letter from the West. The pretender styles the Basileus
# "Rex Graecorum" and signs himself Emperor of the Romans.
###########################################################
basileus_two_emperors.0001 = {
	type = character_event
	window = basileus_imperial_confrontation
	title = basileus_two_emperors.0001.t
	desc = {
		# A Latin-speaking Basileus (eastern_roman / western_roman / renamed
		# "Roman") is sneered at as a Greek AND has his Latin revival mocked.
		first_valid = {
			triggered_desc = {
				trigger = { culture = { has_cultural_pillar = language_latin } }
				desc = basileus_two_emperors.0001.desc.latin
			}
			desc = basileus_two_emperors.0001.desc
		}
	}
	theme = emperor
	override_background = ep3_constantinople

	left_portrait = {
		character = root
		animation = anger
	}
	right_portrait = {
		character = scope:rival
		animation = writing
	}

	immediate = {
		if = {
			limit = { exists = title:e_hre.holder }
			title:e_hre.holder = { save_scope_as = rival }
		}
		if = {
			limit = { exists = character:70523 }
			character:70523 = { save_scope_as = constantine }
		}
	}

	option = {
		name = basileus_two_emperors.0001.a
		# The chancery is summoned to draft a fitting reply.
	}
}

###########################################################
# 0020 â€” The Demand for Precedence. The Western emperor must choose: concede
# the Basileus's seniority, or refuse. Root = the e_hre holder (AI).
###########################################################
basileus_two_emperors.0020 = {
	type = character_event
	window = basileus_imperial_confrontation
	title = basileus_two_emperors.0020.t
	desc = basileus_two_emperors.0020.desc
	theme = emperor
	override_background = { reference = throne_room }

	left_portrait = {
		character = root
		animation = thinking
	}
	right_portrait = {
		character = scope:basileus
		animation = personality_bold
	}

	immediate = {
		if = {
			limit = { exists = global_var:basileus_two_emperors_challenger }
			global_var:basileus_two_emperors_challenger = { save_scope_as = basileus }
		}
	}

	# Concede precedence â€” acknowledge the senior Roman Emperor.
	option = {
		name = basileus_two_emperors.0020.a
		ai_chance = {
			base = 35
			modifier = { add = 25  scope:basileus = { prestige_level >= 4 } }
			modifier = { add = 20  has_trait = content }
			modifier = { add = 20  has_trait = craven }
			modifier = { add = 15  has_trait = humble }
			modifier = { add = 15  global_var:basileus_two_emperors_heat >= 2  global_var:basileus_two_emperors_heat <= 3 }
			modifier = { add = 20  is_at_war = yes }
			modifier = { add = -25  global_var:basileus_two_emperors_heat >= 5 }
			modifier = { add = -30  OR = { has_trait = ambitious  has_trait = brave  has_trait = arrogant } }
		}
		if = {
			limit = { exists = scope:basileus }
			scope:basileus = { trigger_event = basileus_two_emperors.0022 }
		}
	}

	# Refuse â€” "I am Emperor by the grace of God and Saint Peter!"
	option = {
		name = basileus_two_emperors.0020.b
		ai_chance = {
			base = 50
			modifier = { add = 30  OR = { has_trait = ambitious  has_trait = brave  has_trait = arrogant } }
			modifier = { add = 25  global_var:basileus_two_emperors_heat >= 5 }
			modifier = { add = -20  scope:basileus = { prestige_level >= 4 } }
			modifier = { add = -15  has_trait = craven }
		}
		if = {
			limit = { exists = scope:basileus }
			scope:basileus = { trigger_event = basileus_two_emperors.0021 }
		}
	}
}
