		# is open to talented commoners (lowborn advancement).
	}

		got_council_position_effect = yes
		if = {
			limit = { root.liege_or_court_owner = { basileus_is_imperial_roman_court_trigger = yes } }
			# Imperial-Roman courts: this seat is the Eparch of the City.
			add_character_flag = is_eparch_candidate
			add_character_flag = eparch_unfireable
			trigger_event = {
				on_action = basileus_eparch_appointed_on_action
				days = 1
			}
		}
		else = {
			# Other governments (e.g. China): vanilla Minister of Works handling â€”
			# grants e_minister_of_works with appointment succession so it is not
			# destroyed on an heirless death.
			got_minister_position_effect = { MINISTER_TITLE = e_minister_of_works  MINISTER_POSITION = minister_works }
		}
		if = {
			limit = {
				has_title = title:k_megas_domestikos
			}
			basileus_return_tagma_command_effect = yes
			remove_character_flag = need_military_outfit
		}
	}

	on_fired_from_position = {
		fired_from_council_position_effect = yes
	}

	on_lose_position = {
		# China: releases e_minister_of_works (no-op for the Roman Eparch, who
		# holds no e_minister title). Eparch cleanup is a no-op outside Rome.
		fired_minister_position_effect = yes
		remove_character_flag = is_eparch_candidate
		remove_character_flag = eparch_unfireable
		if = {
			limit = { has_court_position = eparch_court_position }
			remove_court_position = eparch_court_position
		}
	}

	portrait_animation = minister_works

	barbershop_data = {
		position = { 1268.0 172.0 }
		click_to_front = no
	}
}

	modifier = {
		name = minister_grand_marshal_modifier
		monthly_influence = 1
		maa_toughness_mult = 0.04
		scale = council_scaled_admin_value
	}

	modifier = { # Minister modifiers
		name = minister_grand_marshal_modifier
		monthly_merit = 1
		scale = council_scaled_celestial_ministry_value
	}

	council_owner_modifier = {
		name = learn_on_the_job_modifier
		martial = 1
		scale = marshal_learn_on_the_job_scale
	}

	council_owner_modifier = {
		name = reliable_house_modifier
		martial = 1
		scale = marshal_reliable_house_scale
	}

	valid_character = {
		exists = root.liege_or_court_owner
		basileus_imperial_court_male_only_trigger = { COURT_OWNER = root.liege_or_court_owner }
		basileus_can_be_a_minister_trigger = { COURT_OWNER = root.liege_or_court_owner  SKILL = martial }
		# In an imperial-autocracy court ONLY, the Megas Domestikos must be a
		# LANDLESS character: courtiers and landless noble-family (d_) holders
		# qualify; landed governors (holding any county-or-higher territorial
		# title) do not â€” appointing them to the landless office would strip their
		# governed themes. Other governments (incl. vanilla China) keep their pool.
		trigger_if = {
			limit = {
				root.liege_or_court_owner = { government_has_flag = government_is_roman_imperial }
			}
			NOT = {
				any_held_title = {
					tier >= tier_county
					is_noble_family_title = no
					# Exempt the office itself â€” once appointed, the Megas Domestikos
					# holds k_megas_domestikos (kingdom tier), and without this exemption
					# they'd fail their own validity check and be kicked next tick.
					NOT = { this = title:k_megas_domestikos }
				}
			}
		}
	}

	on_get_position = {
		got_council_position_effect = yes
		if = {
			limit = { root.liege_or_court_owner = { basileus_is_imperial_roman_court_trigger = yes } }
			# Imperial-Roman courts: the Megas Domestikos commands the standing tagma
			# (on its own k_megas_domestikos title) and wears military gear.
			basileus_grant_tagma_command_effect = yes
			add_character_flag = need_military_outfit
		}
		else = {
			# Other governments (e.g. China): vanilla Grand Marshal handling â€” grants
			# e_minister_grand_marshal with appointment succession so it is not
			# destroyed on an heirless death.
			got_minister_position_effect = { MINISTER_TITLE = e_minister_grand_marshal  MINISTER_POSITION = minister_grand_marshal }
		}
	}

	on_fired_from_position = {
		fired_from_council_position_effect = yes
		remove_character_flag = need_military_outfit
	}

	on_lose_position = {
		# Roman path (self-gated, safe everywhere): return the command title + gear.
		basileus_return_tagma_command_effect = yes
		remove_character_flag = need_military_outfit
		# Non-Roman path (e.g. China): release the e_minister title via the vanilla
		# minister-fired handler (no-op for the Roman Megas Domestikos).
		if = {
			limit = { NOT = { root.liege_or_court_owner = { basileus_is_imperial_roman_court_trigger = yes } } }
			fired_minister_position_effect = yes
		}
	}

	portrait_animation = marshal

	barbershop_data = {
		position = { 918.0 172.0 }
		click_to_front = no
	}
}
