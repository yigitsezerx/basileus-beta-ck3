# =====================================================================
# BASILEUS — ROMAN IMPERIAL GOVERNMENT
# ---------------------------------------------------------------------
# A reskin/variant of administrative_government for the Byzantine /
# restored-Roman empire. It deliberately keeps mechanic_type = administrative
# and the government_is_administrative flag, so ALL existing TGP plumbing
# (ministries, treasury, legitimacy, domicile, obligations) keeps working
# unchanged. The distinguishing flag government_is_roman_imperial is what the
# mod's meritocracy code targets, so no other administrative realm is affected.
#
# Differences from administrative_government:
#   * own name / adjective / colour (loc + this file)
#   * extra flag government_is_roman_imperial
#   * small "flavour" character_modifier bonuses on top of admin's
#   * NOT is_mechanic_type_default (administrative_government stays the default)
#
# Attached to e_byzantium / h_roman_empire / h_eastern_roman_empire holders at
# game start and on title gain (see basileus_government_on_actions.txt).
# =====================================================================

roman_imperial_government = {
	government_rules = {
		create_cadet_branches = yes
		rulers_should_have_dynasty = yes
		dynasty_named_realms = no
		administrative = yes
		landless_playable = yes
		legitimacy = yes
		state_faith = yes
		use_as_base_on_landed = yes
		use_as_base_on_rank_up = yes
		allow_as_base_for_baronies = no
		inherit_from_dynastic_government = no
		sticky_government = yes
		house_aspirations = yes
		noble_families = yes
		admin_allows_holding_multiple_primary_tier_titles = yes
		treasury = yes
		replace_gold_cost_by_treasury = yes
		allow_accolades = yes
	}

	royal_court = any
	blocked_subject_courts = { roman_imperial_government meritocratic_government celestial_government japan_administrative_government administrative_government steppe_admin_government }

	domicile_type = estate
	main_administrative_tier = duchy
	min_appointment_tier = duchy
	minimum_provincial_maa_tier = duchy
	# Which titles participate in the administrative title men-at-arms system.
	#   top_vassals_and_top_liege = the emperor's DIRECT vassals' titles (of any
	#   tier, INCLUDING the kingdom-tier landless k_megas_domestikos) + the top
	#   liege's primary. This is what TFE ships for its kingdom-tier comes_domesticorum
	#   ministry, and it is what surfaces the Imperial Tagmata in the Military window.
	#   Trade-off (engine limitation, no single enum avoids it): the emperor's
	#   DIRECTLY-held duchies do NOT get theme cards under this setup — appoint
	#   governors and their duchies appear as themes instead (intended admin play).
	administrative_title_maa_setup = top_vassals_and_top_liege

	fallback = 3

	primary_holding = castle_holding
	valid_holdings = { city_holding temple_citadel_holding }
	required_county_holdings = { castle_holding city_holding church_holding }

	vassal_contract_group = basileus_roman_imperial_vassal

	can_get_government = {
		# Mirror administrative_government's LAAMP Construct Holding guard.
		trigger_if = {
			limit = {
				OR = {
					has_variable = city_construction_is_temple_citadel
					has_variable = city_construction_is_castle
				}
			}
			always = no
		}
	}

	character_modifier = {
		levy_size = -0.5
		feudal_government_tax_contribution_mult = -0.5
		feudal_government_levy_contribution_mult = -0.5
		clan_government_tax_contribution_mult = -0.5
		clan_government_levy_contribution_mult = -0.5
		men_at_arms_cap = -1
		men_at_arms_limit = -1
		knight_limit = -5
		vassal_limit = 100
		accolades = -1
		title_creation_cost_mult = -0.5
		mercenary_hire_cost_mult = 2
		men_at_arms_maintenance = 0.5

		# Vassals get Treasury income that is this percentage less than their taxes to their liege
		monthly_treasury_from_liege_mult = -0.15

		# --- Roman Imperial flavour: a merit-selected bureaucracy serves more
		# loyally and lends the court dignity. (Modest, tunable.) ---
		councillor_opinion = 10
		court_grandeur_baseline_add = 1
	}

	top_liege_character_modifier = {
		# 85% of vassal income goes to the treasury
		monthly_treasury_from_vassals = 0.85
		men_at_arms_maintenance = 1.5
		men_at_arms_recruitment_cost = 1
	}

	# Use flags instead of has_government for moddability if possible (i.e., wherever not visible to the player).
	flags = {
		government_is_administrative
		government_is_roman_imperial

		# These flags only exist for loc and in-game descriptions
		government_has_influence
		government_has_treasury
		government_has_title_men_at_arms
		government_has_powerful_families
		government_is_settled
		government_uses_domicile_but_not_adventurer
		government_uses_domain_limit
		government_uses_admin_province_obligations
	}

	mechanic_type = administrative
	# NOTE: deliberately NOT is_mechanic_type_default — administrative_government
	# remains the default for the administrative mechanic.

	color = { 85 20 34 }
	realm_mask_offset = { 0.0 0.050 }
	realm_mask_scale = { 0.95 0.95 }
}
