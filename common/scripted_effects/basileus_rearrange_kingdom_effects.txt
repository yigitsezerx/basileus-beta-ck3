# =====================================================================
# BASILEUS — REARRANGE THE KINGDOMS (kingdom-level de jure reshaping)
# ---------------------------------------------------------------------
# Called by basileus_rearrange_kingdoms_decision. Where the themes decision
# reshapes DUCHY-level de jure borders, this reshapes KINGDOM-level borders:
# it pulls a set of duchies under one surviving kingdom and destroys the
# emptied parent kingdoms. The survivor is then optionally renamed (rename
# event) and its duchies recolored into a harmonized family.
#
# Adding a widget = one basileus_rearrange_kingdom_X_effect + one
# basileus_recolor_X_duchies_effect + one lock flag + one decision item + loc.
#
# root = the decision-taking emperor.
# =====================================================================

# --- SHARED: grant a formed title to the reorganizing emperor -------
# Gives $TITLE$ to root, but ONLY if it is currently unheld (matches the Bavaria
# guard). Idempotent: a safe no-op when the forming effect already granted it, so
# it can be called uniformly after every restructuring widget. root = the emperor.
basileus_grant_title_to_root_if_unheld_effect = {
	if = {
		limit = { NOT = { exists = $TITLE$.holder } }
		create_title_and_vassal_change = {
			type = created
			save_scope_as = btgr_change
			add_claim_on_loss = no
		}
		$TITLE$ = {
			change_title_holder = {
				holder = root
				change = scope:btgr_change
			}
		}
		resolve_title_and_vassal_change = scope:btgr_change
	}
}

# --- ANATOLIA -------------------------------------------------------
# Consolidates every de jure duchy of k_nikaea + k_anatolia + k_pontus under
# k_anatolia, then destroys the now-empty k_nikaea and k_pontus. The 8 duchies
# already de jure under k_anatolia are harmless no-ops; only the 6 from nikaea
# and pontus actually move. Membership verified against 00_landed_titles.txt:
#   k_nikaea = { d_opsikion d_optimatoi }
#   k_pontus = { d_paphlagonia d_bucellaria d_armeniac d_chaldia }
# so both kingdoms empty out cleanly and are safe to destroy.
basileus_rearrange_kingdom_anatolia_effect = {
	# 1. Move the 8 SURVIVING duchies under k_anatolia. (The four merge themes are
	#    a hard prerequisite, so d_cappadocia/d_charsianon/d_mesopotamia/d_chaldia/
	#    d_paphlagonia/d_bucellaria no longer exist by the time this runs.)
	title:d_thracesia   = { set_de_jure_liege_title = title:k_anatolia }
	title:d_opsikion    = { set_de_jure_liege_title = title:k_anatolia }
	title:d_optimatoi   = { set_de_jure_liege_title = title:k_anatolia }
	title:d_cibyrrhaeot = { set_de_jure_liege_title = title:k_anatolia }
	title:d_anatolia    = { set_de_jure_liege_title = title:k_anatolia }
	title:d_sebasteia   = { set_de_jure_liege_title = title:k_anatolia }
	title:d_armeniac    = { set_de_jure_liege_title = title:k_anatolia }
	title:d_cilicia     = { set_de_jure_liege_title = title:k_anatolia }
	title:d_bucellaria    = { set_de_jure_liege_title = title:k_anatolia }
	title:d_paphlagonia     = { set_de_jure_liege_title = title:k_anatolia }
	title:d_chaldia     = { set_de_jure_liege_title = title:k_anatolia }
	title:d_cappadocia     = { set_de_jure_liege_title = title:k_anatolia }
	title:d_charsianon     = { set_de_jure_liege_title = title:k_anatolia }
	title:d_mesopotamia    = { set_de_jure_liege_title = title:k_anatolia }

	# 2. Keep the survivor de jure under the empire (defensive; usually a no-op).
	title:k_anatolia = { set_de_jure_liege_title = title:e_byzantium }

	# 3. Defensive guard: re-home any county still sitting DIRECTLY under the
	#    soon-destroyed kingdoms (none expected — they only held duchies — but
	#    this prevents an orphan if the data ever changes).
	title:k_nikaea = {
		every_in_de_jure_hierarchy = {
			limit = {
				tier = tier_county
				de_jure_liege = title:k_nikaea
			}
			set_de_jure_liege_title = title:k_anatolia
		}
	}
	title:k_pontus = {
		every_in_de_jure_hierarchy = {
			limit = {
				tier = tier_county
				de_jure_liege = title:k_pontus
			}
			set_de_jure_liege_title = title:k_anatolia
		}
	}

	# 4. Destroy the emptied kingdoms.
	destroy_title = title:k_nikaea
	destroy_title = title:k_pontus

	# 5. Lock the widget.
	add_to_global_variable_list = {
		name = unavailable_unique_decisions
		target = flag:kingdom_anatolia_rearranged
	}
}

# --- ANATOLIA SPLIT (Eastern / Western) -----------------------------
# SHELVED with event basileus_rearrange_kingdoms.0002 (the "sundering"): this
# effect and basileus_recolor_eastern_anatolia_and_duchies_effect below are no
# longer invoked. Kept defined for easy un-shelving (restore .0002 to call them).
# Follow-up to the unification: carve k_anatolia into a western survivor
# (k_anatolia, 5 duchies) and a new eastern kingdom (k_eastern_anatolia, 9
# duchies), with three border-cleanup county moves and new capitals. The new
# kingdom is granted to the emperor (root) until the grant event resolves; both
# halves are renamed culture-dynamically. root = the decision-taking emperor.
basileus_split_anatolia_effect = {
	# 1. Border cleanup — move the three counties to their new duchies FIRST,
	#    so they ride along to the correct kingdom when the duchies are reparented.
	title:c_pamphylia = { set_de_jure_liege_title = title:d_cilicia }    # -> east
	title:c_pisidia   = { set_de_jure_liege_title = title:d_thracesia }  # -> west
	title:c_pacatiana = { set_de_jure_liege_title = title:d_thracesia }  # -> west

	# 2.
	title:k_eastern_anatolia = { set_de_jure_liege_title = title:e_byzantium }
	title:d_anatolia    = { set_de_jure_liege_title = title:k_eastern_anatolia }
	title:d_cappadocia    = { set_de_jure_liege_title = title:k_eastern_anatolia }
	title:d_sebasteia   = { set_de_jure_liege_title = title:k_eastern_anatolia }
	title:d_armeniac    = { set_de_jure_liege_title = title:k_eastern_anatolia }
	title:d_cilicia     = { set_de_jure_liege_title = title:k_eastern_anatolia }
	title:d_charsianon    = { set_de_jure_liege_title = title:k_eastern_anatolia }
	title:d_mesopotamia    = { set_de_jure_liege_title = title:k_eastern_anatolia }
	title:d_chaldia    = { set_de_jure_liege_title = title:k_eastern_anatolia }

	# 3. Capitals (set AFTER the duchy moves so each county is de jure in-realm).
	title:k_eastern_anatolia = { set_capital_county = title:c_trebizond }
	title:k_anatolia         = { set_capital_county = title:c_ionia }

	# 4. Grant the new kingdom to the emperor for now (mirrors the Thrace
	#    elevation). The follow-up grant event may hand it to a candidate.
	if = {
		limit = { NOT = { exists = title:k_eastern_anatolia.holder } }
		create_title_and_vassal_change = {
			type = created
			save_scope_as = ea_split_change
			add_claim_on_loss = no
		}
		title:k_eastern_anatolia = {
			change_title_holder = {
				holder = root
				change = scope:ea_split_change
			}
		}
		resolve_title_and_vassal_change = scope:ea_split_change
	}

	# 5. Rename both halves, culture-dynamic on the emperor (greek vs else).
	#    set_title_name bakes a fixed string (its _adj sibling supplies the
	#    adjective) — mirrors basileus_set_theme_name_effect.
	if = {
		limit = { culture = culture:greek }
		title:k_anatolia         = { set_title_name = BASILEUS_WEST_ANATOLIA_NAME_GREEK }
		title:k_eastern_anatolia = { set_title_name = BASILEUS_EAST_ANATOLIA_NAME_GREEK }
	}
	else = {
		title:k_anatolia         = { set_title_name = BASILEUS_WEST_ANATOLIA_NAME }
		title:k_eastern_anatolia = { set_title_name = BASILEUS_EAST_ANATOLIA_NAME }
	}
}

# Grant k_eastern_anatolia to $HOLDER$ and make them a real, self-contained vassal
# king: they receive the kingdom title, the land the EMPEROR personally holds in
# the east, AND the existing eastern governors are re-pointed as THEIR vassals.
# Mirrors basileus_grant_tagma_command_effect's grant pattern. Scope: root = emperor.
#
# NOTE: re-vassalizing appointed governors under a king-tier vassal in admin
# government is non-standard — verify in-game (right government, themes intact, no
# appointment glitches). If it misbehaves, drop the second vassal-change block.
basileus_grant_eastern_anatolia_effect = {
	$HOLDER$ = { save_scope_as = ea_new_king }

	# --- 1. Hand the kingdom + the emperor's directly-held eastern land to the king. ---
	create_title_and_vassal_change = {
		type = granted
		save_scope_as = ea_grant_change
		add_claim_on_loss = no
	}
	title:k_eastern_anatolia = {
		change_title_holder = {
			holder = scope:ea_new_king
			change = scope:ea_grant_change
		}
	}
	title:k_eastern_anatolia = {
		every_de_jure_county = {
			limit = { holder = root }
			change_title_holder = {
				holder = scope:ea_new_king
				change = scope:ea_grant_change
			}
		}
		every_in_de_jure_hierarchy = {
			limit = { tier = tier_duchy  holder = root }
			change_title_holder = {
				holder = scope:ea_new_king
				change = scope:ea_grant_change
			}
		}
	}
	resolve_title_and_vassal_change = scope:ea_grant_change

	# --- 2. Re-vassalize the existing eastern governors under the new king. ---
	# Any de jure duchy/county within k_eastern_anatolia still held by one of the
	# emperor's vassals (a governor) becomes the new king's vassal instead. Runs
	# AFTER the kingdom is resolved to the king, so he can be their liege.
	create_title_and_vassal_change = {
		type = granted
		save_scope_as = ea_vassal_change
		add_claim_on_loss = no
	}
	title:k_eastern_anatolia = {
		every_in_de_jure_hierarchy = {
			limit = {
				OR = { tier = tier_duchy  tier = tier_county }
				exists = holder
				holder = { is_vassal_of = root }
				NOT = { holder = root }
				NOT = { holder = scope:ea_new_king }
			}
			holder = {
				change_liege = {
					liege = scope:ea_new_king
					change = scope:ea_vassal_change
				}
			}
		}
	}
	resolve_title_and_vassal_change = scope:ea_vassal_change

	# Keep the new king on the realm's Roman Imperial (admin) government.
	scope:ea_new_king = {
		if = {
			limit = { NOT = { has_government = roman_imperial_government } }
			change_government = roman_imperial_government
		}
	}
}

# --- FRANCOGALLIA: merge Aquitaine into France ----------------------
# Optional follow-up to basileus_francogallia_effect (which forms BOTH k_france
# and k_aquitaine). If the emperor chooses to merge (event .0010 option A), every
# de jure duchy under k_aquitaine is reparented to k_france and the now-empty
# southern kingdom is destroyed. NO recolor — the absorbed southern duchies keep
# their existing wine shades. Mirrors the Anatolia unify de-jure-loop + destroy.
basileus_merge_aquitaine_into_france_effect = {
	if = {
		limit = { exists = title:k_aquitaine }
		title:k_aquitaine = {
			every_in_de_jure_hierarchy = {
				limit = { tier = tier_duchy }
				set_de_jure_liege_title = title:k_france
			}
			# Defensive: any county still directly under k_aquitaine.
			every_in_de_jure_hierarchy = {
				limit = {
					tier = tier_county
					de_jure_liege = title:k_aquitaine
				}
				set_de_jure_liege_title = title:k_france
			}
		}
		# Keep the survivor de jure under the empire (defensive; usually a no-op).
		title:k_france = { set_de_jure_liege_title = title:e_byzantium }
		destroy_title = title:k_aquitaine
	}
}
