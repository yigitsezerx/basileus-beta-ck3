# Senate of Magnaura — initialisation, succession reset, yearly approval pulse.
# Approval is stored as a scripted variable `senate_approval` on title:e_byzantium,
# range 0–100, default 50.

on_game_start_after_lobby = {
	on_actions = {
		basileus_senate_init
	}
}

on_title_gain = {
	on_actions = {
		basileus_senate_on_title_gain
	}
}

on_title_lost = {
	on_actions = {
		basileus_senate_on_title_lost
	}
}

# Senate-sanctioned war: when the Basileus declares an offensive war against
# a serious target with the senate's approval, apply a 5-year blessing.
# Vanilla scope: scope:attacker, scope:defender, scope:war.
on_war_started = {
	on_actions = {
		basileus_senate_on_war_started
	}
}

yearly_global_pulse = {
	on_actions = {
		basileus_senate_yearly_pulse
		basileus_senate_ai_appoint
	}
}

# ---------------------------------------------------------------
# AI auto-appointment of senate dignities.
# ---------------------------------------------------------------
# AI rulers with senate access never staffed the Senate on their own, leaving
# their seats empty and their approval permanently low (the "debuff for no
# reason"). Each year, every AI senate-capable ruler fills ONE free seat per
# tier (Hypatos / Senator / Patrician) with the best eligible vassal, so the
# Senate gradually fills and approval recovers. Gated on prestige so the
# appointment cost (handled in on_court_position_received) doesn't bankrupt them.
# Works for e_byzantium AND restored-empire (h_roman_empire/h_eastern_roman_empire)
# holders via basileus_can_use_senate_trigger.
# ---------------------------------------------------------------
basileus_senate_ai_appoint = {
	effect = {
		every_ruler = {
			limit = {
				is_ai = yes
				basileus_can_use_senate_trigger = yes
				prestige >= 100
			}
			save_scope_as = senate_lord

			# Hypatos — single seat.
			if = {
				limit = { NOT = { any_court_position_holder = { type = hypatus_court_position } } }
				ordered_vassal = {
					limit = { basileus_senate_ai_candidate_trigger = yes }
					order_by = diplomacy
					max = 1
					scope:senate_lord = {
						appoint_court_position = {
							recipient = prev
							court_position = hypatus_court_position
						}
					}
				}
			}
			# Senators — up to 5 seats; fill one per year.
			if = {
				limit = { NOT = { any_court_position_holder = { type = senator_court_position  count >= 5 } } }
				ordered_vassal = {
					limit = { basileus_senate_ai_candidate_trigger = yes }
					order_by = diplomacy
					max = 1
					scope:senate_lord = {
						appoint_court_position = {
							recipient = prev
							court_position = senator_court_position
						}
					}
				}
			}
			# Patricians — up to 5 seats; fill one per year.
			if = {
				limit = { NOT = { any_court_position_holder = { type = patrician_court_position  count >= 5 } } }
				ordered_vassal = {
					limit = { basileus_senate_ai_candidate_trigger = yes }
					order_by = diplomacy
					max = 1
					scope:senate_lord = {
						appoint_court_position = {
							recipient = prev
							court_position = patrician_court_position
						}
					}
				}
			}
		}
	}
}

# Note: no on_character_culture_change hook needed — the basileus/emeritus
# traits switch their displayed name dynamically via the trait's name_block,
# so culture changes are reflected automatically without any script action.

# ---------------------------------------------------------------
# Initialise senate_approval on the empire title at game start.
# ---------------------------------------------------------------
basileus_senate_init = {
	effect = {
		# Initialise for every senate-capable ruler (e_byzantium AND restored
		# empires). Approval is stored on each ruler's PRIMARY (imperial) title,
		# matching what the senate-window GUI reads via GetPrimaryTitle.
		every_ruler = {
			limit = { basileus_can_use_senate_trigger = yes }
			primary_title = {
				if = {
					limit = { NOT = { exists = var:senate_approval } }
					set_variable = { name = senate_approval value = 50 }
				}
				# Trend tracking var — same default as approval so first-year
				# tooltip shows "no change" instead of a wild delta from 0.
				if = {
					limit = { NOT = { exists = var:senate_approval_last_year } }
					set_variable = { name = senate_approval_last_year value = var:senate_approval }
				}
			}
			basileus_senate_recompute_counts = yes
			# Sync the threshold modifier to current approval (covers the case
			# where approval was modified via console / outside the pulse).
			basileus_senate_apply_threshold_modifiers = yes
		}
	}
}

# ---------------------------------------------------------------
# Demote any basileus traits when e_byzantium is lost.
# scope:title = the title that was lost; root = the character who lost it.
# ---------------------------------------------------------------
basileus_senate_on_title_lost = {
	effect = {
		if = {
			limit = {
				OR = {
					scope:title = title:e_byzantium
					scope:title = title:h_roman_empire
					scope:title = title:h_eastern_roman_empire
				}
				is_alive = yes
			}
			basileus_senate_revert_ascended_traits = yes
			# Strip approval / dynasty-dominance modifiers from the
			# ex-emperor — these are tied to the throne, not the person.
			remove_character_modifier = senate_approval_high_modifier
			remove_character_modifier = senate_approval_low_modifier
			remove_character_modifier = senate_dynasty_dominance_modifier
		}
	}
}

# ---------------------------------------------------------------
# Reset senate_approval to 50 whenever e_byzantium changes hands.
# ---------------------------------------------------------------
basileus_senate_on_title_gain = {
	effect = {
		if = {
			limit = {
				OR = {
					scope:title = title:e_byzantium
					scope:title = title:h_roman_empire
					scope:title = title:h_eastern_roman_empire
				}
			}
			# Senate blesses succession — if the OUTGOING reign ended with
			# approval >= 70, apply the Anointed modifier to the new holder
			# BEFORE the approval reset. The check is on the title var still
			# holding the old value at this moment.
			if = {
				limit = {
					scope:title = { var:senate_approval >= 70 }
				}
				add_character_modifier = {
					modifier = senate_anointed_modifier
					years = 5
				}
			}

			scope:title = {
				set_variable = {
					name = senate_approval
					value = 50
				}
				# Initialise the last-year tracking var for the trend display.
				set_variable = {
					name = senate_approval_last_year
					value = 50
				}
			}
			# Clear inherited threshold modifiers on the new emperor —
			# they'll be re-evaluated by the yearly pulse, and dynasty-
			# dominance is also re-checked here so the new emperor doesn't
			# inherit a stale dominance modifier from a prior holder.
			remove_character_modifier = senate_approval_high_modifier
			remove_character_modifier = senate_approval_low_modifier
			remove_character_modifier = senate_dynasty_dominance_modifier
			# Refresh count variables — court positions transfer on succession,
			# but our cached counts need to be re-synced for the new holder.
			basileus_senate_recompute_counts = yes

			# If the new holder previously served in the senate, ascend their
			# tenure/emeritus trait(s) to the `_basileus` variant.
			basileus_senate_ascend_senate_traits = yes

			# Re-evaluate dynasty dominance immediately for the new holder so
			# the modifier reflects current senate composition without waiting
			# for the next yearly pulse.
			basileus_senate_check_dynasty_dominance = yes

			# Apply the high/low approval modifier immediately based on the
			# (just-reset-to-50) approval. At 50 this clears both, but the call
			# is here so any future change to the on-gain approval is handled.
			basileus_senate_apply_threshold_modifiers = yes
		}
	}
}

# ---------------------------------------------------------------
# Yearly recalculation of senate_approval. Relaxed tuning (Mar 2026)
# — gentle climb when staffed, soft floor when empty.
#
# Baseline yearly drift:   -3  (mild institutional drag)
# Per filled Hypatos:      +1.5   (halved to +0.75 if approval >= 80)
# Per filled Senator:      +1.0   (halved to +0.5  if approval >= 80)
# Per filled Patrician:    +0.5   (halved to +0.25 if approval >= 80)
# Per senate member with opinion < 0:    -0.25 (no soft cap, softened
#                                       from -0.5 — mild displeasure
#                                       shouldn't bite hard)
# Per senate member with opinion >= 50:  +0.5  (unchanged)
#
# Net result at fully-staffed (1 Hypatos + 5 Senators + 5 Patricians):
#   Approval <80, mixed opinions:   ~+6 / year (was +4)
#   Approval <80, all friendly:     ~+11 / year (was +9)
#   Approval >=80, mixed opinions:  ~+1 / year (was -1; soft cap is now
#                                   a true plateau, not a slow drift down)
#   Approval >=80, all friendly:    ~+6 / year (was +4)
#   Empty senate:                   ~-3 / year (was -5)
# ---------------------------------------------------------------
basileus_senate_yearly_pulse = {
	effect = {
		# Run for every senate-capable ruler (e_byzantium AND restored empires).
		# scope:senate_emperor = the ruler; scope:senate_realm_title = their primary
		# (imperial) title, where approval lives. The body below references these
		# saved scopes in place of the old hardcoded scope:senate_realm_title.
		every_ruler = {
			limit = { basileus_can_use_senate_trigger = yes }
			save_scope_as = senate_emperor
			primary_title = { save_scope_as = senate_realm_title }

			# Trend tracking — snapshot the START-of-year approval so the
			# tooltip can show "vs last year" once this pulse's changes apply.
			# (Stored before any approval mutation in this pulse.)
			scope:senate_realm_title = {
				if = {
					limit = { NOT = { exists = var:senate_approval_last_year } }
					set_variable = { name = senate_approval_last_year value = var:senate_approval }
				}
				set_variable = {
					name = senate_approval_last_year
					value = var:senate_approval
				}
			}

			# Tenure XP: +4 per year served. Tracks auto-progress when thresholds
			# are crossed (10 → Veteran, 25 → Elder).
			scope:senate_emperor = {
				every_court_position_holder = {
					type = hypatus_court_position
					add_trait_xp = { trait = hypatos_tenure value = 4 }
				}
				every_court_position_holder = {
					type = senator_court_position
					add_trait_xp = { trait = senator_tenure value = 4 }
				}
				every_court_position_holder = {
					type = patrician_court_position
					add_trait_xp = { trait = patrician_tenure value = 4 }
				}
			}

			# Resync ascended-trait variants on the current emperor AND every
			# courtier (former senate members usually stay at the imperial
			# court). Catches culture changes the on_action might have missed
			# and migrates pre-existing wrongly-tagged traits from older saves.
			# The resync effect is a no-op for characters without any senate
			# trait, so iterating the whole court is cheap.
			scope:senate_emperor = {
				basileus_senate_resync_ascended_variant = yes
				every_courtier = {
					basileus_senate_resync_ascended_variant = yes
				}
			}

			# Dynasty dominance check — flag if any dynasty has ≥3 senate members.
			basileus_senate_check_dynasty_dominance = yes

			# Safety net: keep count variables in sync with reality.
			basileus_senate_recompute_counts = yes

			scope:senate_realm_title = {
				change_variable = {
					name = senate_approval
					subtract = 3
				}
			}
			scope:senate_emperor = {
				save_scope_as = senate_emperor

				every_court_position_holder = {
					type = hypatus_court_position
					# Hypatos yearly contribution. Soft cap: above 80 approval,
					# halve the gain (senate runs out of new ways to love you).
					if = {
						limit = { scope:senate_realm_title = { var:senate_approval < 80 } }
						scope:senate_realm_title = { change_variable = { name = senate_approval add = 1.5 } }
					}
					else = {
						scope:senate_realm_title = { change_variable = { name = senate_approval add = 0.75 } }
					}
					if = {
						limit = {
							opinion = { target = scope:senate_emperor value < 0 }
						}
						scope:senate_realm_title = {
							change_variable = {
								name = senate_approval
								subtract = 0.25
							}
						}
					}
					if = {
						limit = {
							opinion = { target = scope:senate_emperor value >= 50 }
						}
						scope:senate_realm_title = {
							change_variable = {
								name = senate_approval
								add = 0.5
							}
						}
					}
				}

				every_court_position_holder = {
					type = senator_court_position
					if = {
						limit = { scope:senate_realm_title = { var:senate_approval < 80 } }
						scope:senate_realm_title = { change_variable = { name = senate_approval add = 1 } }
					}
					else = {
						scope:senate_realm_title = { change_variable = { name = senate_approval add = 0.5 } }
					}
					if = {
						limit = {
							opinion = { target = scope:senate_emperor value < 0 }
						}
						scope:senate_realm_title = {
							change_variable = {
								name = senate_approval
								subtract = 0.25
							}
						}
					}
					if = {
						limit = {
							opinion = { target = scope:senate_emperor value >= 50 }
						}
						scope:senate_realm_title = {
							change_variable = {
								name = senate_approval
								add = 0.5
							}
						}
					}
				}

				every_court_position_holder = {
					type = patrician_court_position
					if = {
						limit = { scope:senate_realm_title = { var:senate_approval < 80 } }
						scope:senate_realm_title = { change_variable = { name = senate_approval add = 0.5 } }
					}
					else = {
						scope:senate_realm_title = { change_variable = { name = senate_approval add = 0.25 } }
					}
					if = {
						limit = {
							opinion = { target = scope:senate_emperor value < 0 }
						}
						scope:senate_realm_title = {
							change_variable = {
								name = senate_approval
								subtract = 0.25
							}
						}
					}
					if = {
						limit = {
							opinion = { target = scope:senate_emperor value >= 50 }
						}
						scope:senate_realm_title = {
							change_variable = {
								name = senate_approval
								add = 0.5
							}
						}
					}
				}
			}

			scope:senate_realm_title = {
				clamp_variable = {
					name = senate_approval
					min = 0
					max = 100
				}
			}

			# Apply / refresh threshold modifiers on the emperor.
			# Tiers: <45 hostile (penalties), 45–69 neutral, ≥70 approving (bonuses).
			basileus_senate_apply_threshold_modifiers = yes
		}
	}
}

# ---------------------------------------------------------------
# Senate-sanctioned war: fires on every war start. If the attacker is
# the Basileus, the target is kingdom-tier+, and senate_approval is
# >= 70, the Basileus gets a 5-year Senate-Sanctioned War modifier.
# Notification toast is sent so the player knows it triggered.
# ---------------------------------------------------------------
basileus_senate_on_war_started = {
	effect = {
		if = {
			limit = {
				scope:attacker = { basileus_can_use_senate_trigger = yes }
				scope:defender.highest_held_title_tier >= tier_kingdom
				scope:attacker.primary_title = { var:senate_approval >= 70 }
			}
			scope:attacker = {
				add_character_modifier = {
					modifier = senate_war_blessed_modifier
					years = 5
				}
				send_interface_toast = {
					type = event_toast_effect_good
					title = senate_war_blessed_toast
				}
			}
		}
	}
}
